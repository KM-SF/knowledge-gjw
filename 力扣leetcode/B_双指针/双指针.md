class Solution(object):
    def maxArea(self, height):
        max_ans = 0
        
        l,r = 0,len(height)-1

        while l<r:
            area = min(height[l], height[r]) * (r-l)
            max_ans = max(max_ans, area)
            if height[r] > height[l]:
                l+=1
            else:
                r-=1
        return max_ans
